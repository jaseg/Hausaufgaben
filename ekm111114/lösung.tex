&=\frac{1}{2}m\dot x^2\\
&=\frac{1}{2}mx_0^2\frac{g}{\ell}e^{2\sqrt{\frac{g}{\ell}}t}\\
&=\frac{mg}{2\ell}x_0^2e^{2\sqrt{\frac{g}{\ell}}t}\\
E_{pot}&=mgh-mg\frac{x}{\ell}h+mg\frac{x}{\ell}\left(h-\frac{x}{2}\right)\\

&=mgh-\frac{mg}{2\ell}x_0^2e^{2\sqrt{\frac{g}{\ell}}t}\\
